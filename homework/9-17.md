    /** 
     * Multiplexor:
     * out = a if sel == 0
     *       b otherwise
     */

    CHIP Mux {
        IN a, b, sel;
        OUT out;

        PARTS:
       // Put your code here:
        Not(in=sel,out=Notsel);
        And(a=a,b=Notsel,out=aNotsel);
        And(a=sel,b=b,out=selb);
        Or(a=aNotsel,b=selb,out=out);
    }
